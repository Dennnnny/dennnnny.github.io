// do post 是之後產生一個連結網址  給你打api的時候
// 這個function 會在你對網址 method:POST 時執行的
function doPost(e) {
  // 拿名字叫 messages 的sheet(工作表)
  const sheet =
    SpreadsheetApp.getActiveSpreadsheet().getSheetByName("messages");

  // 拿參數
  // 這邊我傳了 user 跟 message 當參數
  const params = JSON.parse(e.postData.contents);
  const { user, message } = params;
  const createTime = NOW();

  // 在表單裡插入一行資料
  // 內容是 [createTime, user, message] 分別儲存在前三欄的儲存格裡
  sheet.appendRow([createTime, user, message]);

  const result = {
    code: 200,
    message: "成功",
  };

  // 最後給api 一個response
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(
    ContentService.MimeType.JSON
  );
}
